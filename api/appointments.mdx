## Overview

The Appointments API allows you to create, manage, and track veterinary appointments with intelligent scheduling and triage integration.

## The Appointment Object

<ResponseField name="id" type="string" required>
  Unique identifier for the appointment
</ResponseField>

<ResponseField name="petId" type="string" required>
  ID of the pet
</ResponseField>

<ResponseField name="petName" type="string" required>
  Pet's name (denormalized for performance)
</ResponseField>

<ResponseField name="ownerId" type="string" required>
  ID of the pet owner
</ResponseField>

<ResponseField name="ownerName" type="string" required>
  Owner's full name
</ResponseField>

<ResponseField name="vetId" type="string" required>
  ID of the assigned veterinarian
</ResponseField>

<ResponseField name="type" type="enum" required>
  Appointment type: `wellness`, `urgent`, `emergency`, `dental`, `surgery`, `grooming`, `follow_up`
</ResponseField>

<ResponseField name="reason" type="string">
  Reason for visit
</ResponseField>

<ResponseField name="symptoms" type="string[]" default="[]">
  List of reported symptoms
</ResponseField>

<ResponseField name="triageLevel" type="enum" required>
  Triage urgency: `emergency`, `urgent`, `routine`, `info`
</ResponseField>

<ResponseField name="scheduledDate" type="string" required>
  Appointment date (YYYY-MM-DD)
</ResponseField>

<ResponseField name="scheduledTime" type="string" required>
  Appointment time (HH:MM)
</ResponseField>

<ResponseField name="durationMinutes" type="number" default="30">
  Expected duration in minutes
</ResponseField>

<ResponseField name="status" type="enum" default="pending">
  Status: `pending`, `scheduled`, `confirmed`, `checked_in`, `in_progress`, `completed`, `cancelled`, `no_show`, `rescheduled`
</ResponseField>

<ResponseField name="notes" type="string">
  Internal appointment notes
</ResponseField>

<ResponseField name="diagnosis" type="string">
  Diagnosis (filled after appointment)
</ResponseField>

<ResponseField name="treatmentNotes" type="string">
  Treatment notes
</ResponseField>

<ResponseField name="prescriptions" type="string[]">
  Prescribed medications
</ResponseField>

<ResponseField name="followUpDate" type="string">
  Follow-up appointment date
</ResponseField>

<ResponseField name="callId" type="string">
  Associated voice call ID (if booked via AI)
</ResponseField>

<ResponseField name="confirmationSent" type="boolean" default="false">
  Whether confirmation was sent
</ResponseField>

<ResponseField name="reminder24hSent" type="boolean" default="false">
  Whether 24h reminder was sent
</ResponseField>

<ResponseField name="reminder2hSent" type="boolean" default="false">
  Whether 2h reminder was sent
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp
</ResponseField>

  -H "Content-Type: application/json" \

    "petName": "Buddy",

    "triageLevel": "urgent",
    "scheduledDate": "2024-12-16",
    "scheduledTime": "14:30",
    "durationMinutes": 45

## Appointment Workflow Example

```typescript
import { appointmentsApi, slotsApi } from '@/lib/api';

// 1. Find available slots
const slots = await slotsApi.search({
  date: '2024-12-16',
  urgency: 'urgent'
});

if (slots.length === 0) {
  console.log('No slots available');
  return;
}

// 2. Book the first available slot
const appointment = await appointmentsApi.create({
  petId: 'pet-123',
  petName: 'Buddy',
  ownerId: 'owner-456',
  ownerName: 'John Smith',
  vetId: slots[0].vetId,
  type: 'urgent',
  reason: 'Vomiting',
  symptoms: ['vomiting', 'lethargy'],
  triageLevel: 'urgent',
  scheduledDate: slots[0].date,
  scheduledTime: slots[0].time,
  durationMinutes: 30
});

console.log('Appointment booked:', appointment.id);

// 3. Update status when client arrives
await appointmentsApi.update(appointment.id, {
  status: 'checked_in'
});

// 4. Mark in progress during examination
await appointmentsApi.update(appointment.id, {
  status: 'in_progress'
});

// 5. Complete with diagnosis and treatment
await appointmentsApi.update(appointment.id, {
  status: 'completed',
  diagnosis: 'Gastroenteritis',
  treatmentNotes: 'Prescribed bland diet and anti-nausea medication',
  prescriptions: ['Cerenia 16mg'],
  followUpDate: '2024-12-23'
});
```
