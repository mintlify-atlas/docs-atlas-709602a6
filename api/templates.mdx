## Overview

The Templates API provides access to customizable SOAP note templates and email templates for consistent documentation and communication.

## SOAP Template Object

<ResponseField name="id" type="string" required>
  Unique template identifier
</ResponseField>

<ResponseField name="title" type="string" required>
  Template display name
</ResponseField>

<ResponseField name="type" type="enum" required>
  Template type: `soap`, `soap_specialist`, `formatted_dictation`, `dental_canine`, `dental_feline`, `radiograph`, `callback`, `tech_appointment`, `surgery_report`, `staff_meeting`, `rounds`, `lecture_note`, `custom`
</ResponseField>

<ResponseField name="description" type="string" required>
  Template description
</ResponseField>

<ResponseField name="detailLevel" type="enum" default="concise">
  Detail level: `concise` or `detailed`
</ResponseField>

<ResponseField name="isDefault" type="boolean" default="false">
  Whether this is the default template
</ResponseField>

<ResponseField name="isBuiltIn" type="boolean" default="false">
  Whether this is a built-in system template
</ResponseField>

<ResponseField name="sections" type="TemplateSection[]" required>
  Array of template sections
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp
</ResponseField>

### TemplateSection Object

<ResponseField name="id" type="string" required>
  Section identifier (used as key in generated content)
</ResponseField>

<ResponseField name="name" type="string" required>
  Section display name
</ResponseField>

<ResponseField name="fields" type="TemplateField[]" required>
  Array of fields in this section
</ResponseField>

<ResponseField name="order" type="number" required>
  Display order
</ResponseField>

### TemplateField Object

<ResponseField name="id" type="string" required>
  Field identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Field display name
</ResponseField>

<ResponseField name="type" type="enum" required>
  Field type: `text`, `number`, `date`, `select`, `checkbox`, `textarea`
</ResponseField>

<ResponseField name="required" type="boolean" default="false">
  Whether field is required
</ResponseField>

<ResponseField name="options" type="string[]">
  Options for select fields
</ResponseField>

<ResponseField name="defaultValue" type="string">
  Default value
</ResponseField>

<ResponseField name="section" type="string">
  Parent section ID
</ResponseField>

<ResponseField name="order" type="number" required>
  Display order within section
</ResponseField>

## Email Templates

### Email Template Object

<ResponseField name="id" type="string" required>
  Template identifier
</ResponseField>

<ResponseField name="name" type="string" required>
  Template name
</ResponseField>

<ResponseField name="subject" type="string" required>
  Email subject line (supports variables)
</ResponseField>

<ResponseField name="body" type="string" required>
  Email body content (supports variables)
</ResponseField>

<ResponseField name="type" type="string" required>
  Template type (appointment_reminder, follow_up, etc.)
</ResponseField>

<ResponseField name="variables" type="string[]" required>
  Available variables for substitution
</ResponseField>

### Using Email Templates

```typescript
const fillEmailTemplate = (
  template: EmailTemplate,
  variables: Record<string, string>
): { subject: string; body: string } => {
  let subject = template.subject;
  let body = template.body;

  // Replace all variables
  for (const [key, value] of Object.entries(variables)) {
    const placeholder = `{${key}}`;
    subject = subject.replace(new RegExp(placeholder, 'g'), value);
    body = body.replace(new RegExp(placeholder, 'g'), value);
  }

  return { subject, body };
};

// Example usage
const reminderTemplate = emailTemplates.find(t => t.id === 'appointment-reminder')!;
const email = fillEmailTemplate(reminderTemplate, {
  petName: 'Buddy',
  ownerName: 'John Smith',
  appointmentDate: 'December 16, 2024',
  appointmentTime: '2:30 PM',
  vetName: 'Dr. Sarah Chen',
  clinicName: 'Paw & Care Veterinary Clinic'
});

console.log(email.subject); // "Reminder: Buddy has an appointment tomorrow"
```
